# perspective_grasp host Python deps
#
# Installed into the workspace venv (<ws>/.venv) by install_host.sh /
# install_dependencies.sh. The venv uses --system-site-packages so apt's
# rclpy / cv_bridge / numpy stay visible — these pins layer on top.
#
# numpy<2 is load-bearing: cv_bridge in ros-jazzy is compiled against the
# NumPy 1.x C ABI, and ultralytics transitively pulls numpy.

numpy<2
ultralytics==8.4.40
# ByteTrack (ultralytics' default tracker) needs lap. Without this pin
# ultralytics auto-pip-installs it on first track() call inside the
# yolo_byte_tracker node — the install succeeds but emits "Restart runtime
# for updates to take effect", and the first session's tracking is unreliable
# until restart. Pinning here makes the install reproducible.
lap>=0.5.12

# torch + torchvision are pinned so a fresh install can never end up with a
# wheel built for a CUDA version newer than the deployed NVIDIA driver
# supports (the "cu130 wheel on a CUDA-12.8 driver" trap — see CLAUDE.md
# anti-pattern p). The cuXXX index is selected at install time by
# install_host_python() via the PERSPECTIVE_TORCH_CUDA env var (default:
# cu126, which matches our Phase 4 Docker baseline). See docs/installation.md
# for the driver-to-cuXXX mapping.
torch==2.6.0
torchvision==0.21.0
